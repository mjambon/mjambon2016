(*
  Polymorphic hash table using physical equality on the keys.

  Released in the Public Domain. Use at your own risk.
  Author: Martin Jambon (January 2009)
*)

type ('a, 'b) t

val add : ('a, 'b) t -> 'b -> 'a -> unit
val clear : ('a, 'b) t -> unit
val copy : ('a, 'b) t -> ('a, 'b) t
val create : int -> ('a, 'b) t
val find : ('a, 'b) t -> 'b -> 'a
val find_all : ('a, 'b) t -> 'b -> 'a list
val fold : ('a -> 'b -> 'c -> 'c) -> ('a, 'b) t -> 'c -> 'c
val iter : ('a -> 'b -> unit) -> ('a, 'b) t -> unit
val length : ('a, 'b) t -> int
val mem : ('a, 'b) t -> 'b -> bool
val remove : ('a, 'b) t -> 'b -> unit
val replace : ('a, 'b) t -> 'b -> 'a -> unit
